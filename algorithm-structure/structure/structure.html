# Queue

  // 链表实现队列

  import java.util.Iterator;

  public class Queue<Item> implements Iterable<Item> {
    private Node first;
    private Node last;
    private int N;
    private class Node {
        Item item;
        Node next;
    }
    public boolean isEmpty() {
        return first == null;
    }
    public int size() {
        return N;
    }
    public void enqueue(Item item) {
        Node oldFirst = last;
        last = new Node();
        last.item = item;
        last.next = null;
        if(isEmpty()) first = last;
        else oldFirst.next = last;
        N++;
    }
    public Item dequeue() {
        Item item = first.item;
        first = first.next;
        // 如果只有一个，需要把 first 也设置为 last
        if(isEmpty()) last = null;
        N--;
        return item;
    }
    public Iterator<Item> iterator() {
        return new ListIterator();
    }
    private class ListIterator implements Iterator<Item> {
        private Node current = first;
        public boolean hasNext() {
            return current != null;
        }
        public void remove() {

        }
        public Item next() {
            Item item = current.item;
            current = current.next;
            return item;
        }
    }
}

# 弹性栈

因为 Java 语言的特性，创建的数据都是固定的，会出行超出或太多空位置，在每次添加或删除时，计算一下当前的长度

import java.util.Iterator;

public class ResizingArrayStack<Item> implements Iterable<Item> {
    private Item[] a = (Item[]) new Object[1];
    private int N = 0;
    public boolean isEmpty() {
        return N == 0;
    }
    public int size() {
        return N;
    }
    private void resize(int max) {
        Item[] temp = (Item[]) new Object[max];
        for (int i = 0; i < N; i++) {
            temp[i] = a[i];
        }
        a = temp;
    }
    public void push(Item item) {
        if(N == a.length) resize(2 * a.length);
        a[N++] = item;
    }
    public Item pop() {
        Item item = a[--N];
        a[N] = null;
        if(N > 0 && N == a.length / 4) resize(a.length / 2);
        return item;
    }
    public Iterator<Item> iterator() {
        return new ReverseArrayIterator();
    }
    private class ReverseArrayIterator implements Iterator<Item> {
        private int i = N;
        public boolean hasNext() { return i > 0; }
        public Item next() { return a[--i]; }
        public void remove() {}
    }
}

# Task

function Hero(name) {
  const _this = this
  this.tasks = []
 
  this.tasks.push(() => console.log(`Hi! This is ${name}`))
 
  this.kill = function (num) {
    const fn = () => console.log(`Kill ${num} ${num > 1 ? 'bugs' : 'bug'}`)
    _this.tasks.push(fn)
    return _this
  }
  this.recover = function (num) {
    const fn = () => console.log(`Recover ${num} ${num > 1 ? 'bloods' : 'blood'}`)
    _this.tasks.push(fn)
    return _this
  }
 
  this.sleep = function (time) {
    const fn = async () => {
      return new Promise(resolve => {
        setTimeout(() => {
          console.log(`Waiting ${time}s`)
          resolve()
        }, time * 1000)
      })
    }
    _this.tasks.push(fn)
    return _this
  }
  const executeTasks = async function () {
    for (const task of _this.tasks) {
      await task()
    }
  }
  executeTasks()
  return this
}
const a = Hero('37er').sleep(1);
