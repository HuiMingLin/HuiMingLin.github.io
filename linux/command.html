# 命令

# 1.ls

列出当前目录中的文件和子目录

ls
1

# 2.cd

切换到指定目录。

cd directory_path
1

# 3.pwd

显示当前所在的工作目录的路径。

ls
1

# 4.mkdir

创建一个新的目录。

mkdir directory_name
1

# 5.rm

删除指定的文件或目录

rm file_name
1

# 6.cp

将文件或目录复制到指定位置。

cp source_file destination_file
1

# 7.mv

将文件或目录移动到指定位置。

mv source_file destination_file
1

# 8.cat

显示文件的全部内容。

cat file_name
1

# 9.less

分页显示文件的内容。

less file_name
1

# 10.head

显示文件的开头部分内容。

head file_name
1

# 11.tail

显示文件的结尾部分内容。

tail file_name
1

# 12.grep

在文件中搜索指定的文本。

grep "search_text" file_name
1

# 25.grep -r -

递归搜索指定目录中的文件内容。

grep -r "search_text" directory_path
1

# 50.grep -i -

在文件中忽略大小写搜索指定的文本。

grep -i "search_text" file_name
1

# 13.find

在指定目录及其子目录中搜索文件。

find directory_path -name "file_name"
1

1

gzip file_name
